
\begin{tikzpicture}[

declare function={

a=4;

b=a/4;

k=.6;

h=3;

yo=b^2/h;

go=asin(yo/b);

xo=a*cos(go);

ax=a*k;

bx=b*k;

},

join=round,

cap=round,

scale=.85

]

% Định nghĩa các điểm

\path

(0:0) coordinate (O)

(90:h) coordinate (S)

(60:a and b) coordinate (B)

(-80:a and b) coordinate (A)

(barycentric cs:A=1,B=1) coordinate (I)

(barycentric cs:S=1,I=3) coordinate (H);

% Các đoạn thẳng

\draw[densely dashed]

(H) -- (O) -- (S) -- (I) -- (O)

(S) -- (B) -- (A) -- (O) -- (B);

% Vẽ cung elip

\draw[gray, densely dotted]

(xo,yo) arc (go:180-go:a and b);

% Vẽ hình kín

\draw

(S) -- (A)

(S) -- (xo,yo) arc (go:-180-go:a and b) -- cycle;

% Vẽ các điểm và nhãn

\foreach \p/\g in {O/-120, S/90, A/-100, B/30, I/-60, H/30}

\fill[ball color=white] (\p) circle (.8pt)

node[shift={(\g:.15)}, scale=.6] {$\p$};

\end{tikzpicture}
