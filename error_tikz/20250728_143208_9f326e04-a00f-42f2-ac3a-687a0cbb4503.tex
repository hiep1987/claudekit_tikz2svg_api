  % Tọa độ chính
  \path 
    (0,0) coordinate (O)
    (120:r) coordinate (A)
    (210:r) coordinate (B)
    (-30:r) coordinate (C)
    
    ($(B)!(A)!(C)$) coordinate (A_0)
    ($(A)!(B)!(C)$) coordinate (B_0)
    ($(A)!(C)!(B)$) coordinate (C_0)

    ($(B)!0.5!(C)$) coordinate (A_1)

    ($(A)!0.5!(B)$) coordinate (C_1)

    ($(B)!0.5!(H)$) coordinate (B_2)

  ;

  \path[name path=BB0] (B) -- (B_0);
  \path[name intersections={of=AA0 and BB0, by=H}];

  % Đường nét chính

  \draw [dashed,blue] (A_1)--(B_1)--(C_1)--cycle;

  \draw [dashed,red] (A_2)--(B_2)--(C_2)--cycle;

    let \p1=($(Op)-(B_1)$) in (Op) circle ({veclen(\x1,\y1)});

  % Vòng lặp đánh nhãn
  \foreach \name/\angle/\label in {
    O/-90/O, A/90/A, B/180/B, C/0/C,

  } {

  }
